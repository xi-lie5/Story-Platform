## 3. 共享组件库（Phase 1 产出）

### 3.1 `utils.js` — 工具函数

```javascript
// 所有页面共享——来自各页面现有实现的合并

function escapeHtml(text) {
  var d = document.createElement('div');
  d.textContent = text || '';
  return d.innerHTML;
}

function showToast(message, type) {
  type = type || 'error';
  var t = document.getElementById('errorToast');
  if (!t) { alert(message); return; }
  t.textContent = message;
  t.style.display = 'block';
  t.style.background = type === 'success' ? '#22c55e' : '#dc2626';
  setTimeout(function() { t.style.display = 'none'; }, 5000);
}

function getAvatar(avatar, name) {
  if (!avatar) return 'https://ui-avatars.com/api/?background=6366f1&color=fff&name=' +
    encodeURIComponent(name || 'U');
  if (/^https?:\/\//i.test(avatar)) return avatar;
  var normalized = avatar.startsWith('/') ? avatar : '/' + avatar;
  return API_CONFIG.BASE_URL + normalized;
}

function formatDate(isoString) {
  return new Date(isoString).toLocaleDateString('zh-CN');
}

function truncate(text, maxLen) {
  maxLen = maxLen || 200;
  if (!text) return '';
  return text.length <= maxLen ? text : text.substring(0, maxLen) + '...';
}
```

### 3.2 `components.js` — 可复用 UI 组件

```javascript
// Loading spinner
function renderLoading(message) {
  message = message || '加载中...';
  return '<div style="text-align:center;padding:3rem;color:#64748b">' +
    '<i class="fas fa-spinner fa-spin" style="font-size:2rem;display:block;margin-bottom:1rem;opacity:0.3"></i>' +
    message + '</div>';
}

// Empty state
function renderEmpty(icon, message, actionText, actionUrl) {
  return '<div style="text-align:center;padding:4rem 1rem;color:#64748b">' +
    '<i class="fas fa-' + (icon || 'inbox') + '" style="font-size:2.5rem;display:block;margin-bottom:1rem;opacity:0.3"></i>' +
    '<p>' + message + '</p>' +
    (actionUrl ? '<a href="' + actionUrl + '" style="display:inline-block;margin-top:1rem;padding:0.5rem 1.5rem;border-radius:8px;background:linear-gradient(135deg,#6366f1,#8b5cf6);color:#fff;text-decoration:none;font-weight:600">' + (actionText || '去创建') + '</a>' : '') +
    '</div>';
}

// Error state with retry
function renderError(message, retryFn) {
  return '<div style="text-align:center;padding:3rem;color:#f87171">' +
    escapeHtml(message) + '<br>' +
    '<button onclick="(' + retryFn.toString() + ')()" style="margin-top:1rem;padding:0.4rem 1rem;border-radius:8px;border:1px solid rgba(255,255,255,0.15);background:rgba(255,255,255,0.04);color:#e2e8f0;cursor:pointer">重试</button>' +
    '</div>';
}

// Pagination
function renderPagination(currentPage, totalPages, goPageFn) {
  if (totalPages <= 1) return '';
  var html = '<div style="display:flex;justify-content:center;gap:0.5rem;margin-top:2rem">';
  html += '<button ' + (currentPage <= 1 ? 'disabled' : '') +
    ' onclick="' + goPageFn + '(' + (currentPage - 1) + ')" style="...">◀</button>';
  for (var p = 1; p <= totalPages; p++) {
    html += '<button class="' + (p === currentPage ? 'active' : '') +
      '" onclick="' + goPageFn + '(' + p + ')" style="...">' + p + '</button>';
  }
  html += '<button ' + (currentPage >= totalPages ? 'disabled' : '') +
    ' onclick="' + goPageFn + '(' + (currentPage + 1) + ')" style="...">▶</button>';
  html += '</div>';
  return html;
}
```

### 3.3 故事卡片组件

所有故事列表页面（browse、ai-stories、my_stories）使用统一卡片结构：

```javascript
function renderStoryCard(story, options) {
  options = options || {};
  var showAuthor = options.showAuthor !== false;
  var showStats = options.showStats !== false;
  var actions = options.actions || '';
  var moods = (story.mood_tags || []).map(function(t) {
    return '<span class="card-tag">' + escapeHtml(t) + '</span>';
  }).join('');

  return '<div class="card">' +
    '<div class="card-title">' + escapeHtml(story.title) + '</div>' +
    (showAuthor ? '<div class="card-author">' +
      '<img src="' + getAvatar(story.author_avatar, story.author_name) + '"> ' +
      escapeHtml(story.author_name || '未知作者') + '</div>' : '') +
    '<div class="card-desc">' + escapeHtml(truncate(story.description || story.start_prompt || '', 200)) + '</div>' +
    (moods ? '<div class="card-tags">' + moods + '</div>' : '') +
    (showStats ? '<div class="card-footer">' +
      '<div class="card-stats">' +
        '<span><i class="fas fa-eye"></i> ' + (story.view_count || 0) + '</span>' +
        '<span><i class="fas fa-star"></i> ' + (story.rating || 0) + '</span>' +
      '</div>' + actions + '</div>' : '') +
    '</div>';
}
```

## 6. 设计检查清单（提交前端 PR 前）

- [ ] 导航栏使用 `navbar.js` 标准结构（不手写导航 HTML）
- [ ] 使用 `utils.js` 中的 `escapeHtml()`，不再自己定义
- [ ] API 调用使用 `API_CONFIG.buildUrl()` + `API_CONFIG.get/post/put/patch/delete()`
- [ ] 认证检查使用 `AuthUI.requireAuth()` 或 `AuthUI.getAuthState()`
- [ ] 按钮有 disabled 态 + loading 态
- [ ] 列表有空态提示（使用 `components.js` 的 `renderEmpty()`）
- [ ] API 调用有 try/catch 错误处理 + 重试按钮（使用 `components.js` 的 `renderError()`）
- [ ] 深色主题 CSS 变量统一（不手写颜色值）
