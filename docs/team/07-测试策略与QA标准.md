### 2.2 每个端点最少测试覆盖

```javascript
// 示例: aiStory.test.js
describe('POST /aiStory/config', function() {
  it('creates a rulebook with valid data', async function() {
    var res = await request(app)
      .post('/api/v1/aiStory/config')
      .set('Authorization', 'Bearer ' + token)
      .send({ title: 'Test', worldSetting: '...', startPrompt: '...', characters: [{name:'X'}] });
    expect(res.status).toBe(201);
    expect(res.body.data.id).toBeDefined();
  });

  it('rejects empty title', async function() {
    var res = await request(app)
      .post('/api/v1/aiStory/config')
      .set('Authorization', 'Bearer ' + token)
      .send({ title: '', worldSetting: '...', startPrompt: '...', characters: [{name:'X'}] });
    expect(res.status).toBe(400);
  });

  it('rejects without auth', async function() {
    var res = await request(app)
      .post('/api/v1/aiStory/config')
      .send({ title: 'Test', worldSetting: '...', startPrompt: '...', characters: [{name:'X'}] });
    expect(res.status).toBe(401);
  });
});
```

### 2.3 测试数据隔离

```javascript
// helpers.js — 每次测试前创建独立用户
async function createTestUser() {
  var ts = Date.now();
  var res = await request(app)
    .post('/api/v1/auth/register')
    .send({ username: 'test_' + ts, email: 'test_' + ts + '@test.com',
            password: 'Test123456', confirmPassword: 'Test123456' });
  return { token: res.body.data.token, userId: res.body.data.userId };
}

// setup.js — 测试结束后清理
afterAll(async function() {
  // 删除所有测试用户及其数据
  await pool.query("DELETE FROM users WHERE username LIKE 'test_%'");
});
```

    // ... 需要真实 API key
  });

## 3. 前端测试

### 3.1 工具函数单元测试（vitest + jsdom）

```javascript
// front/tests/utils.test.js
describe('escapeHtml', function() {
  it('escapes < and >', function() {
    expect(Utils.escapeHtml('<script>')).toBe('&lt;script&gt;');
  });
  it('returns empty string for null', function() {
    expect(Utils.escapeHtml(null)).toBe('');
  });
});

describe('truncate', function() {
  it('truncates long text', function() {
    expect(Utils.truncate('1234567890', 5)).toBe('12345...');
  });
  it('returns full short text', function() {
    expect(Utils.truncate('hi', 5)).toBe('hi');
  });
});
```

### 3.2 前端手动检查清单

提交前端 PR 前，在浏览器中验证：

| 检查项 | 方法 |
|--------|------|
| 所有页面导航栏一致 | 逐个打开 14 个页面，目视检查 |
| 浏览器 Console 无红色错误 | F12 → Console |
| Network 无 4xx/5xx 错误 | F12 → Network → 过滤 status ≥ 400 |
| 移动端导航菜单可展开/收起 | Chrome DevTools → 手机模式 → 点击汉堡菜单 |
| 加载态显示 | 刷新页面 → 观察加载动画 |
| 空态显示 | 新账号登录 → 查看列表页空态提示 |
| 错误态显示 + 重试按钮 | 断开服务器 → 刷新页面 → 确认错误消息和重试按钮可用 |
| 表单验证 | 提交空表单 → 观察错误提示 |

## 4. CI Pipeline（GitHub Actions）

```yaml
# .github/workflows/ci.yml
name: CI
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    services:
      mysql:
        image: mysql:8.0
        env:
          MYSQL_ROOT_PASSWORD: testpassword
          MYSQL_DATABASE: story_platform
        ports:
          - 3306:3306
        options: >-
          --health-cmd "mysqladmin ping"
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '18'

      - name: Install dependencies
        run: cd backend && npm ci

      - name: Run migrations
        run: cd backend && node db/migrate.js
        env:
          MYSQL_HOST: 127.0.0.1
          MYSQL_USER: root
          MYSQL_PASSWORD: testpassword
          MYSQL_DATABASE: story_platform
          JWT_SECRET: test_secret_key_min_32_characters_long
          JWT_REFRESH_SECRET: test_refresh_secret_min_32_chars

      - name: Run tests
        run: cd backend && npx vitest run
        env:
          MYSQL_HOST: 127.0.0.1
          MYSQL_USER: root
          MYSQL_PASSWORD: testpassword
          MYSQL_DATABASE: story_platform
          JWT_SECRET: test_secret_key_min_32_characters_long
          JWT_REFRESH_SECRET: test_refresh_secret_min_32_chars
          RUN_AI_TESTS: false
```

## 6. 代码审查清单

提交 PR 时，审查者逐项确认：

### 后端 PR
- [ ] 新端点有对应的测试用例
- [ ] 路由有 `express-validator` 输入验证
- [ ] 错误使用 `next(errorFormat(...))` 返回
- [ ] 写操作使用事务（如涉及多表）
- [ ] 含 LIMIT/OFFSET 的查询使用 `conn.query()` 而非 `conn.execute()`
- [ ] 迁移文件随代码一起提交（如涉及数据库变更）
- [ ] 新端点已更新 API 契约文档

### 前端 PR
- [ ] 使用 `assets/js/utils.js` 中的共享函数（不重复定义）
- [ ] 使用 `assets/js/navbar.js` 标准导航栏结构
- [ ] API 调用使用 `API_CONFIG.buildUrl()` + 标准方法
- [ ] 有 loading 态、empty 态、error 态 + retry
- [ ] 颜色/间距引用 CSS 变量（不手写色值）
- [ ] `escapeHtml()` 处理了所有用户输入
